    # 镜像常用操作

    # 1、列出镜像

    docker images

    # 2、拉取镜像

    docker pull node:8.10.0

    # 3、搜索镜像

    docker search node

    # 4、导出导入镜像

    docker save node > node.tar
docker load < node.tar

    # Docker 容器常用操作

    # 1、创建并启动容器

    docker run -i -t node /bin/bash

    -i启动容器后打开标准输入 -t分配一个终端，与服务器建立一个会话

    # 2、列出容器

    docker ps

    # 3、进入运行中的容器

    docker attach 容器ID

    # 4、执行命令

    docker exec -i -t 容器ID ls -l

    # 5、停止容器

    docker stop 容器ID

    # 6、终止容器

    docker kill 容器ID

    # 7、启动容器

    docker start 容器ID

    # 8、重启容器

    docker restart 容器ID

    # 9、删除容器

    docker rm 容器ID

    # 10、导入导出容器

    docker export 容器ID > node.tar
docker import node.tar cyy/node:latest
